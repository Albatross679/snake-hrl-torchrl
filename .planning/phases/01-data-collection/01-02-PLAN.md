**1. Add `_respawn_worker` function** (new, above `_multiprocess_collect`):
```python
def _respawn_worker(worker_id, config, shared_counter, worker_counter, nan_counter,
                    shutdown_event, existing_episode_offset, respawn_count, save_dir):
    """Respawn a dead/stalled worker with a fresh seed."""
    # Clean up any .tmp files from the dead worker
    prefix = f"batch_w{worker_id:02d}"
    for tmp in save_dir.glob(f"{prefix}_*.tmp"):
        tmp.unlink(missing_ok=True)

    # Reset the worker's per-worker counter (it tracks this worker's progress)
    with worker_counter.get_lock():
        worker_counter.value = 0

    # New seed offset: large prime * respawn_count to avoid seed collision
    seed_offset = respawn_count * 7919

    p = mp.Process(
        target=_worker_fn,
        args=(worker_id, config, shared_counter, worker_counter, nan_counter,
              shutdown_event, existing_episode_offset, seed_offset),
        name=f"collector-{worker_id}-r{respawn_count}",
    )
    p.start()
    return p
```

**3. Enhance the per-worker status check** in the monitoring loop (inside the `for i, (p, wc) in enumerate(...)` loop from Plan 01):

After computing `delta` (Plan 01 already computes `current - prev_worker_counts[i]`):
```python
if p.is_alive():
    if delta == 0:
        stall_counts[i] += 1
    else:
        stall_counts[i] = 0

    # Stall detection: alive but zero progress for N consecutive intervals
    if stall_counts[i] >= config.stall_intervals:
        log_event(event_log_path, "worker_stalled", "warn", worker_id=i,
                  details={"stall_intervals": stall_counts[i], "pid": p.pid})
        _send_alert(wandb_run, "Worker Stalled",
                    f"Worker {i} (PID {p.pid}) stalled for {stall_counts[i]} intervals. Restarting.",
                    level="WARN", wait_duration=config.alert_wait_duration)
        p.terminate()
        p.join(timeout=10)
        respawn_counts[i] += 1
        workers[i] = _respawn_worker(i, config, shared_counter, worker_counters[i],
                                      nan_counters[i], shutdown_event,
                                      existing_episode_offset, respawn_counts[i], save_dir)
        stall_counts[i] = 0
        log_event(event_log_path, "worker_respawned", "info", worker_id=i,
                  details={"respawn_count": respawn_counts[i], "reason": "stall"})
else:
    # Worker died
    exitcode = p.exitcode
    log_event(event_log_path, "worker_died", "error", worker_id=i,
              details={"exitcode": exitcode, "pid": p.pid})
    _send_alert(wandb_run, "Worker Died",
                f"Worker {i} (PID {p.pid}) exited with code {exitcode}. Respawning.",
                level="ERROR", wait_duration=config.alert_wait_duration)
    respawn_counts[i] += 1
    workers[i] = _respawn_worker(i, config, shared_counter, worker_counters[i],
                                  nan_counters[i], shutdown_event,
                                  existing_episode_offset, respawn_counts[i], save_dir)
    stall_counts[i] = 0
    log_event(event_log_path, "worker_respawned", "info", worker_id=i,
              details={"respawn_count": respawn_counts[i], "reason": "death"})
```

```python
def _send_alert(wandb_run, title: str, text: str, level: str = "WARN", wait_duration: int = 300):
    """Send W&B alert if wandb is active. Silently no-op otherwise."""
    if wandb_run is None:
        return
    try:
        from wandb import AlertLevel
        level_map = {"INFO": AlertLevel.INFO, "WARN": AlertLevel.WARN, "ERROR": AlertLevel.ERROR}
        import wandb
        wandb.alert(
            title=title[:63],  # W&B title limit is 64 chars
            text=text,
            level=level_map.get(level, AlertLevel.WARN),
            wait_duration=wait_duration,
        )
    except Exception:
        pass  # Never crash collection on alert failure
```

**Create `aprx_model_elastica/monitor.py`** -- a standalone external observer that reads the JSONL event log and displays per-worker status. This is the `python -m aprx_model_elastica monitor` command per the locked decision.

The monitor is a thin reader -- it does NOT manage workers, send alerts, or write to W&B. It provides visibility into an ongoing collection run by tailing `events.jsonl`.

```python
"""External monitor: reads event log and shows per-worker status.

Usage:
    python -m aprx_model_elastica monitor --save-dir data/surrogate

Tails the events.jsonl file written by the collector and displays
a live per-worker status table updated every poll interval.
"""
```

**Core functionality:**

1. `def tail_events(save_dir: str, poll_interval: float = 30.0) -> None`
   - Main loop: reads `{save_dir}/events.jsonl`, tracks file position with `f.tell()`/`f.seek()`.
   - Maintains `worker_status: dict[int, dict]` with keys: `status` (alive/dead/stalled), `transitions`, `delta`, `last_event_time`, `respawn_count`, `nan_discards`.
   - Each poll: read new lines since last position, update status dict, print status table, sleep.
   - Handle missing file gracefully (print "Waiting for events.jsonl..." and retry).
   - Handle partial/corrupt JSON lines with try/except (skip them).
   - Event type handlers:
     - `worker_status`: update status, transitions, delta
     - `worker_died`: set status to "dead"
     - `worker_respawned`: set status to "alive", increment respawn_count
     - `worker_stalled`: set status to "stalled"
     - `nan_discard`: increment nan_discards for that worker
     - `shutdown_requested`: print shutdown notice
     - `shutdown_complete`: print completion and exit

2. `def _print_status_table(worker_status: dict, elapsed: float) -> None`
   - Clears terminal (print `\033[2J\033[H` ANSI escape for clear+home).
   - Prints header: `"Collection Monitor -- {elapsed:.0f}s elapsed"`
   - Prints table:
     ```
     Worker | Status  | Transitions | Delta/poll | Respawns | NaN Discards
     -------|---------|-------------|------------|----------|-------------
       0    | alive   |    12,345   |     890    |    0     |      2
       1    | dead    |     5,678   |       0    |    1     |      0
     ```
   - Also prints collection summary: total transitions across all workers, total NaN discards.

3. `def main() -> None`
   - Parse args: `--save-dir` (required), `--poll-interval` (default 30.0)
   - Call `tail_events(save_dir, poll_interval)`
