#!/usr/bin/env bash
# Phase 02.2: RL-step-only surrogate data collection (steps_per_run=1)
# Collects (state, action, next_state) pairs with exactly 1 RL step per reset.
# Data saved to data/surrogate_rl_step/ (separate from Phase 02.1 data/surrogate/).

cd "$(dirname "$0")/../.."

python3 -m aprx_model_elastica collect \
  --steps-per-run 1 \
  --save-dir data/surrogate_rl_step \

  --num-transitions 50000000 \
  --episodes-per-save 100 \
  --perturbation-fraction 0.3 \
  --perturb-omega-std 1.5 \
  --sobol \
  --seed 43 \
  "$@"
