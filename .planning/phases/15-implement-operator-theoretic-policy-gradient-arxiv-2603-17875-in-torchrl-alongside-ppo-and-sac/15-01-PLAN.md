```python
@dataclass
class OTPGConfig(RLConfig):
    """Operator-Theoretic Policy Gradient configuration.

    Based on MM-RKHS algorithm (Gupta & Mahajan, 2026, arXiv:2603.17875).
    Adapts the majorization-minimization framework to continuous action
    spaces with neural network function approximation.

    Loss = -E[ratio * A] + beta * MMD^2(pi_new, pi_old) + (1/eta) * KL + value_coef * critic_loss
    """
    # Majorization bound coefficient (beta in paper Eq 6.1)
    beta: float = 1.0

    # Mirror descent step size (eta_k in paper Eq 7.1)
    eta: float = 1.0

    # MMD kernel configuration
    mmd_kernel: str = "rbf"
    mmd_bandwidth: float = 1.0
    mmd_num_samples: int = 16

    # GAE (shared with PPO)
    gae_lambda: float = 0.95
    normalize_advantage: bool = True

    # Critic loss weight
    value_coef: float = 0.5

    # Learning rate schedule
    lr_schedule: str = "linear"
    lr_end: float = 1e-5

    # Early stopping
    patience_batches: int = 200
```

No entropy_coef field (per CONTEXT.md: "No entropy bonus -- the KL regularizer and MMD penalty handle exploration; no entropy_coef term").

   ```python
   def _update(self, batch: TensorDict) -> Dict[str, float]:
       metrics = {"loss_policy": 0.0, "loss_critic": 0.0, "mmd_penalty": 0.0,
                  "kl_divergence": 0.0, "grad_norm": 0.0, "policy_entropy": 0.0}
       actual_updates = 0

       for epoch in range(self.config.num_epochs):
           indices = torch.randperm(batch.numel())
           num_batches = max(1, batch.numel() // self.config.mini_batch_size)

           for i in range(num_batches):
               start = i * self.config.mini_batch_size
               end = min((i + 1) * self.config.mini_batch_size, batch.numel())
               mb = batch[indices[start:end]]

               # Forward pass: get new distribution
               dist = self.actor.get_dist(mb)
               log_prob_new = dist.log_prob(mb["action"])
               log_prob_old = mb["sample_log_prob"]

               # Clamp log-ratio for numerical stability, then exp
               log_ratio = (log_prob_new - log_prob_old).clamp(-20.0, 20.0)
               ratio = log_ratio.exp()

               # Advantage (already computed by GAE in train())
               advantage = mb["advantage"]
               if self.config.normalize_advantage:
                   advantage = (advantage - advantage.mean()) / (advantage.std() + 1e-8)

               # Surrogate advantage (no clipping -- trust region via MMD)
               surr_advantage = (ratio * advantage).mean()

               # MMD penalty between old and new policy
               mmd = self._compute_mmd_penalty(mb, dist)

               # KL regularizer (approximate via log-ratio mean)
               kl = log_ratio.mean()

               # Critic loss
               value_pred = self.critic(mb)["state_value"]
               value_target = mb["value_target"]
               critic_loss = F.mse_loss(value_pred, value_target)

               # Policy entropy for diagnostics
               entropy = dist.entropy().mean()

               # Total loss: -surr_advantage + beta*MMD + (1/eta)*KL + value_coef*critic
               policy_loss = -surr_advantage + self.config.beta * mmd + (1.0 / self.config.eta) * kl
               total_loss = policy_loss + self.config.value_coef * critic_loss

               # NaN guard
               if not torch.isfinite(total_loss):
                   print(f"  [WARNING] NaN/inf loss, skipping update")
                   continue

               self.optimizer.zero_grad()
               total_loss.backward()
               grad_norm = nn.utils.clip_grad_norm_(self.loss_params, self.config.max_grad_norm)

               if not torch.isfinite(grad_norm):
                   print(f"  [WARNING] NaN/inf gradients, skipping update")
                   self.optimizer.zero_grad()
                   continue

               self.optimizer.step()
               actual_updates += 1

               metrics["loss_policy"] += policy_loss.item()
               metrics["loss_critic"] += critic_loss.item()
               metrics["mmd_penalty"] += mmd.item()
               metrics["kl_divergence"] += kl.item()
               metrics["grad_norm"] += float(grad_norm)
               metrics["policy_entropy"] += entropy.item()

       for key in metrics:
           metrics[key] /= max(1, actual_updates)
       return metrics
   ```

8. Add `_compute_mmd_penalty()` method:
   ```python
   def _compute_mmd_penalty(self, mb: TensorDict, new_dist) -> torch.Tensor:
       """Compute MMD^2 between old and new policy using linear-time RBF estimator."""
       sigma = self.config.mmd_bandwidth
       n_samples = self.config.mmd_num_samples

       # Sample from new policy (differentiable)
       y = new_dist.rsample((n_samples,))  # [n_samples, batch, act_dim]

       # Sample from old policy (non-differentiable)
       with torch.no_grad():
           old_dist = self.actor.get_dist(mb.clone())
           x = old_dist.sample((n_samples,))  # [n_samples, batch, act_dim]

       # Linear-time unbiased MMD^2 estimator
       n = n_samples // 2
       x1, x2 = x[:n], x[n:2*n]
       y1, y2 = y[:n], y[n:2*n]

       def rbf_kernel(a, b):
           diff = (a - b).pow(2).sum(-1)
           return (-diff / (2 * sigma ** 2)).exp()

       k_xx = rbf_kernel(x1, x2)
       k_yy = rbf_kernel(y1, y2)
       k_xy = rbf_kernel(x1, y1)

       mmd2 = (k_xx + k_yy - 2 * k_xy).mean()
       return mmd2.clamp(min=0.0)  # MMD^2 should be non-negative
   ```
