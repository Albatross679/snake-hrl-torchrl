\section{Introduction}
\label{sec:intro}

Reinforcement learning for soft and continuum robots is bottlenecked by the cost of first-principles simulation.
A snake robot modeled as a Cosserat rod requires resolving elastic body dynamics, anisotropic ground friction, and internally generated muscle moments at every simulation substep --- PyElastica \citep{Naughton2021} uses 500 Position Verlet substeps per control step, costing ${\sim}46$\,ms per environment step.
This report presents a neural surrogate $f_\theta \approx T$ that replaces the transition operator with a single GPU forward pass (${\sim}0.001$\,ms), targeting a four-to-five order-of-magnitude speedup for RL training.

The project has three pillars:
\begin{enumerate}
  \item \textbf{Neural surrogate} --- parallel data collection from 16 PyElastica workers, followed by MLP training on state-delta prediction with per-element CPG phase encoding (\Cref{sec:surrogate}).
  \item \textbf{Direct RL} --- PPO on the Elastica backend (\Cref{sec:rl-elastica}) and SAC/PPO/OTPG on the DisMech backend for the Choi2025 follow task (\Cref{sec:rl-dismech}).
  \item \textbf{Physics-informed models} --- standard PINN, DD-PINN, and physics-regularized surrogate variants that embed the governing equations into training (\Cref{sec:pinn}).
\end{enumerate}

Two physics backends are investigated --- PyElastica (explicit symplectic) and DisMech (implicit backward Euler) --- providing complementary perspectives on surrogate fidelity and RL control.

\subsection{Elastica Physics}
\label{sec:surrogate:physics}

\subsubsection{Cosserat Rod Formulation}
\label{sec:surrogate:physics:cosserat}

The system is fully specified by the tuple $\mathcal{S} = (\mathcal{L},\; \Omega,\; \text{BC},\; \text{IC},\; \boldsymbol{\theta})$:

\begin{table}[H]
  \centering
  \small
  \renewcommand{\arraystretch}{1.3}
  \caption{Well-posedness tuple for the Elastica Cosserat rod system.}
  \label{tab:system-tuple}
  \begin{tabular}{clp{8.5cm}}
    \toprule
    Component & Name & Specification \\
    \midrule
    $\mathcal{L}$ & Governing equations &
      Cosserat rod PDEs: conservation of linear and angular momentum with linear elastic constitutive law \\
    $\Omega$ & Domain &
      $s \in [0, L]$, $L = 0.5$\,m (arc length) $\times$ $t \in [0, T]$ (time) \\
    BC & Boundary conditions &
      Free--free: $\mathbf{F} = \mathbf{M} = \mathbf{0}$ at $s = 0$ and $s = L$ \\
    IC & Initial conditions &
      Straight rod at rest: $\mathbf{x}(s,0) = \mathbf{x}_0 + s\hat{\mathbf{e}}_x$,\; $\mathbf{v} = \boldsymbol{\kappa} = \boldsymbol{\omega} = \mathbf{0}$ \\
    $\boldsymbol{\theta}$ & Parameters &
      $E{=}10^5$\,Pa, $\nu{=}0.5$, $\rho{=}1200$\,kg/m$^3$, $R{=}0.02$\,m, $c_t{=}0.01$, $c_n{=}0.05$, $\gamma{=}0.002$ \\
    Control & CPG forcing &
      $\mathbf{a} = (A, f, k, \phi_0, b) \in \mathbb{R}^5$ enters as time-varying rest curvature $\boldsymbol{\kappa}_0(s,t)$ \\
    \bottomrule
  \end{tabular}
\end{table}

The governing PDEs are:
\begin{subequations}
\label{eq:cosserat-pdes}
\begin{alignat}{2}
  \rho A \,\frac{\partial \mathbf{v}}{\partial t}
    &= \frac{\partial \mathbf{F}}{\partial s} + \mathbf{f}_{\text{ext}}
    &\qquad& \text{(linear momentum)}
  \label{eq:pde-linear} \\[4pt]
  \rho \mathbf{I} \,\frac{\partial \boldsymbol{\omega}}{\partial t}
    &= \frac{\partial \mathbf{M}}{\partial s}
      + \mathbf{m}_{\text{ext}}
      + \frac{\partial \mathbf{x}}{\partial s} \times \mathbf{F}
    &\qquad& \text{(angular momentum)}
  \label{eq:pde-angular}
\end{alignat}
\end{subequations}
with constitutive closure:
\begin{equation}
  \mathbf{F} = \mathbf{S}\,(\boldsymbol{\varepsilon} - \boldsymbol{\varepsilon}_0), \quad
  \mathbf{M} = \mathbf{B}\,(\boldsymbol{\kappa} - \boldsymbol{\kappa}_0), \quad
  \mathbf{S} = \operatorname{diag}(GA, GA, EA), \quad
  \mathbf{B} = \operatorname{diag}(EI, EI, GJ)
  \label{eq:constitutive}
\end{equation}

Spatial discretization on a staggered grid ($N{=}21$ nodes, $N_e{=}20$ elements) reduces the PDE to 124 coupled first-order ODEs:
\begin{equation}
  \dot{\mathbf{s}} = \mathbf{g}(\mathbf{s},\, \mathbf{a},\, t), \qquad
  \mathbf{s} \in \mathbb{R}^{124},\;\; \mathbf{a} \in \mathbb{R}^{5}
  \label{eq:ode-system}
\end{equation}

Position Verlet ($\Delta t = 0.001$\,s, $n_{\text{sub}} = 500$ substeps per RL step of $\Delta t_{\text{ctrl}} = 0.5$\,s) produces the transition operator:
\begin{equation}
  \mathbf{s}_{t+1} = T(\mathbf{s}_t,\, \mathbf{a}_t)
    = \underbrace{V_{\Delta t}(\,\cdot\,;\, \boldsymbol{\kappa}^{\text{rest},499}) \circ \cdots \circ V_{\Delta t}(\,\cdot\,;\, \boldsymbol{\kappa}^{\text{rest},0})}_{500 \text{ substeps, each with distinct rest curvature}}(\mathbf{s}_t)
  \label{eq:transition-operator}
\end{equation}

See \Cref{app:physics} for detailed PDE derivations, staggered grid operators, external force handling, and the DisMech formulation.

\subsubsection{Elastica Transition Operator}
\label{sec:surrogate:physics:algorithm}

\Cref{alg:elastica-transition} expands the full transition operator.

\begin{algorithm}[ht!]\small
\caption{Elastica Transition Operator $T(\mathbf{s}_t,\, \mathbf{a}_t,\, t_{\text{step}}) \to \mathbf{s}_{t+1}$}\label{alg:elastica-transition}
\begin{algorithmic}[1]
\Require State $\mathbf{s}_t \in \mathbb{R}^{124}$, action $\mathbf{a}_t \in \mathbb{R}^5$, step time $t_{\text{step}}$
\Ensure Next state $\mathbf{s}_{t+1} \in \mathbb{R}^{124}$
\State $\mathbf{s} \gets \mathbf{s}_t$
\For{$n = 0, 1, \ldots, 499$}
  \State $t^n \gets t_{\text{step}} + n \cdot \Delta t$
  \State Update rest curvature: $\kappa_v^{\text{rest}} \gets A \sin(2\pi k \cdot s_v + 2\pi f \cdot t^n + \phi_0) + b$ for $v = 1, \ldots, 19$
  \State Verlet half-step: $\mathbf{q}^{1/2} \gets \mathbf{q} + \tfrac{1}{2}\Delta t\;\dot{\mathbf{q}}$
  \State Compute geometry: edge vectors, tangents, strain, curvature
  \State Compute internal forces (shear--stretch) and torques (bending)
  \State Compute external forces: RFT friction + damping + gravity
  \State Compute accelerations: $\ddot{\mathbf{x}} \gets (\mathbf{F}^{\text{int}} + \mathbf{F}^{\text{ext}}) / \mathbf{m}$
  \State Verlet velocity update + full-step position update
\EndFor
\State \textbf{return} $\mathbf{s}_{t+1} = (\mathbf{q},\, \dot{\mathbf{q}})$
\end{algorithmic}
\end{algorithm}

\noindent
\textit{Cost:} ${\sim}377{,}000$ FLOPs/step (${\sim}46$\,ms on CPU).
The CPG phase advances by $\Delta\phi = 2\pi f \Delta t$ per substep --- the surrogate must learn the \textit{integrated effect} of the traveling wave across 500 distinct curvature configurations.

% ---- 3.2 Data Collection ----
\subsection{Data Collection}
\label{sec:surrogate:data}

Training data is generated by 16 independent PyElastica worker processes:

\begin{enumerate}
  \item Actions sampled from 5D CPG space using Sobol quasi-random sequences (50\% Sobol, 50\% uniform).

  \item Each worker stores transitions $(\mathbf{s}_t, \mathbf{a}_t, \mathbf{s}_{t+1}, t_{\text{start}})$ to batch files.
  \item Health monitor respawns stalled workers.
\end{enumerate}

\end{equation}

\subsubsection{Per-Element CPG Phase Encoding}
\label{sec:surrogate:model:phase}

For element $e$, the phase triplet is:
\begin{equation}
  \boldsymbol{\phi}_e
    = \bigl(\sin(2\pi f \cdot t_e),\;
            \cos(2\pi f \cdot t_e),\;
            \kappa_e^{\text{actual}}\bigr), \qquad
  t_e = \frac{k \cdot s_e}{f}
  \label{eq:per-element-phase}
\end{equation}

This provides the exact phase of the serpenoid wave at each element's location, plus the instantaneous curvature $\kappa_e^{\text{actual}}$ for the control error.
Phase features are computed on-the-fly from stored action and $t_{\text{start}}$.

An earlier V1 global phase encoding ($\mathbb{R}^2$) achieved only $R^2 = 0.23$ for angular velocity; the V2 per-element encoding significantly improved prediction.

\subsubsection{Architecture and Prediction}
\label{sec:surrogate:model:arch}

The surrogate predicts a normalized state delta:
\begin{equation}
  f_\theta(\mathbf{z}_t) \approx T(\mathbf{s}_t, \mathbf{a}_t) - \mathbf{s}_t, \qquad
  \hat{\mathbf{s}}_{t+1}
    = \mathbf{s}_t
      + \boldsymbol{\sigma}_\Delta \odot f_\theta(\mathbf{z}_t)
      + \boldsymbol{\mu}_\Delta
  \label{eq:next-state-reconstruction}
\end{equation}

Network architecture:
\begin{equation}
  \mathbb{R}^{189}
    \xrightarrow{\text{Linear}}
    \bigl[\text{Linear}(512) \to \text{LayerNorm} \to \text{SiLU}\bigr]^{\times 3}
    \xrightarrow{\text{Linear}}
    \mathbb{R}^{124}
  \label{eq:architecture}
\end{equation}

\begin{itemize}
  \item Output layer initialized to zero (default prediction: mean empirical transition)
  \item Total trainable parameters: ${\sim}659$K
  \item Delta prediction concentrates the target distribution and provides an implicit residual connection
\end{itemize}

\section{Physics-Informed Neural Network}
\label{sec:pinn}

\subsection{Mathematical Formulation}
\label{sec:pinn:formulation}

\subsubsection{Standard PINN}
\label{sec:pinn:standard}

Physics-informed neural networks \citep{Raissi2019} augment the data-driven loss with a physics residual derived from the governing equations.
Given the state-space model $\dot{\mathbf{x}}(t) = \fssm(\mathbf{x}(t), \mathbf{u}(t))$, a network $\mathbf{h}_\theta$ predicts $\xhat_t = \mathbf{h}_\theta(\mathbf{x}_0, \mathbf{u}, t)$.
The PINN loss:
\begin{multline}
  \mathcal{L}(\theta)
    = \underbrace{\lambda_{\mathrm{IC}} \bigl\| \xhat_0 - \mathbf{x}_0 \bigr\|^2}_{\mathcal{L}_{\mathrm{IC}}}
    + \underbrace{\lambda_{\mathrm{phys}} \frac{1}{N_c}\sum_{k=1}^{N_c}
      \left\| \frac{\partial \xhat_{t_k}}{\partial t} - \fssm(\xhat_{t_k}, \mathbf{u}_{t_k}) \right\|^2}_{\mathcal{L}_{\mathrm{phys}}}
    + \underbrace{\lambda_{\mathrm{data}} \frac{1}{N_d}\sum_{k=1}^{N_d}
      \left\| \xhat_{t_k} - \mathbf{x}_{\mathrm{data},k} \right\|^2}_{\mathcal{L}_{\mathrm{data}}}
  \label{eq:pinn:loss}
\end{multline}

The time derivative $\partial \xhat_t / \partial t$ is obtained via autodiff through the network, doubling the computational cost per collocation point.

\paragraph{Practical bottlenecks:}
\begin{itemize}
  \item \textbf{Expensive autodiff} --- additional backward pass per collocation point
  \item \textbf{Loss balancing} --- three competing objectives require careful weighting
  \item \textbf{Training instability} --- can diverge for systems with $m \gg 1$ coupled states
\end{itemize}

\subsubsection{Physics Regularizer}
\label{sec:pinn:regularizer}

The simplest physics-informed approach adds algebraic constraints as soft regularizers.
Four constraints are enforced via trapezoidal integration on the 124-dimensional state delta $\Delta\mathbf{x} = \mathbf{x}_{t+1} - \mathbf{x}_t$:

\begin{table}[H]
  \centering
  \small
  \renewcommand{\arraystretch}{1.3}
  \caption{Physics regularizer constraints.}
  \label{tab:pinn-constraints}
  \begin{tabular}{@{}clp{7.5cm}@{}}
    \toprule
    \textbf{ID} & \textbf{Constraint} & \textbf{Formulation} \\
    \midrule
    C1 & Translational kinematics & $\Delta x_i \approx (v_{x,i} + \tfrac{1}{2}\Delta v_{x,i})\Delta t$ \\
    C2 & Angular kinematics & $\Delta\psi_e \approx (\omega_{z,e} + \tfrac{1}{2}\Delta\omega_{z,e})\Delta t$ \\
    C3 & Curvature--moment & Spatial gradient consistency ($0.1\times$ weight) \\
    C4 & Bounded kinetic energy & $\text{ReLU}(|\Delta\text{KE}| - E_{\text{thr}})^2$ \\
    \bottomrule
  \end{tabular}
\end{table}

Total regularization: $\mathcal{L}_{\mathrm{reg}} = \mathcal{L}_{\mathrm{kin},x} + \mathcal{L}_{\mathrm{kin},y} + \mathcal{L}_{\mathrm{ang}} + \mathcal{L}_{\mathrm{curv}} + \mathcal{L}_{\mathrm{energy}}$.

\subsubsection{DD-PINN Method}
\label{sec:pinn:ddpinn}

The domain-decoupled PINN \citep{Krauss2024,Licher2025} resolves all three PINN bottlenecks: \textbf{remove time as a network input} and output parameters of an analytically differentiable ansatz.

\begin{equation}
  \boxed{\xhat_t = \mathbf{g}(\avec, t) + \mathbf{x}_0, \qquad \avec = \fnn(\mathbf{x}_0, \mathbf{u}_0;\, \theta)}
  \label{eq:pinn:ddpinn-core}
\end{equation}

The ansatz for each state variable $j$:
\begin{equation}
  g_j(t) = \sum_{i=1}^{n_g} \alpha_{ij}\, e^{-\delta_{ij} t}
    \bigl[\phig(\beta_{ij}\, t + \gamma_{ij}) - \phig(\gamma_{ij})\bigr]
  \label{eq:pinn:ansatz}
\end{equation}
where $\phig(x) = \sin(x)$ and the subtraction ensures $g_j(0) = 0$ exactly.

\paragraph{Key advantages over standard PINN:}
\begin{itemize}
  \item \textbf{Exact IC:} $\mathbf{g}(\avec, 0) \equiv \mathbf{0}$ by construction
  \item \textbf{Closed-form gradients:} $\dot{\mathbf{g}}$ computed analytically --- no autodiff cost
  \item \textbf{Simplified loss:} only 2 terms ($\mathcal{L}_{\mathrm{phys}} + \mathcal{L}_{\mathrm{data}}$)
  \item \textbf{More collocation points:} $10$--$25\times$ more than standard PINNs due to cheap evaluation
\end{itemize}

\begin{figure}[H]
\centering
\resizebox{\textwidth}{!}{%
\begin{tikzpicture}[
    >=Stealth,
    block/.style={draw, rounded corners=3pt, minimum height=1cm, minimum width=2cm,
                  align=center, font=\footnotesize},
    arrow/.style={->, thick},
    lbl/.style={font=\scriptsize, text=gray!70!black}
]

\subsubsection{Limitations of PINNs}
\label{sec:pinn:limitations}

PINNs face fundamental limitations for complex dynamical systems:
\begin{itemize}
  \item \textbf{Spectral bias:} Networks learn low-frequency components first, making high-frequency CPG dynamics difficult.
  \item \textbf{Scaling to high dimensions:} Standard PINNs struggle for $m \gg 1$ (the snake robot has $m = 124$).
  \item \textbf{Discontinuous physics:} Contact and friction forces create non-smooth dynamics that PINNs cannot easily enforce.
  \item \textbf{Ansatz design:} DD-PINNs require an appropriate ansatz function designed a priori for the dynamics class.
\end{itemize}

\subsubsection{Differentiable Cosserat Rod Right-Hand Side}
\label{sec:pinn:cosserat-rhs}

A differentiable implementation of $\fssm$ computes $\dot{\mathbf{x}} = \fssm(\mathbf{x})$ for the 124-dimensional state vector using an inextensible rod approximation (bending + RFT friction, no stretching).
The right-hand side assembles six blocks:
\begin{equation}
  \fssm(\mathbf{x}) = \bigl(\dot{x}_i,\; \dot{y}_i,\; \dot{v}_{x,i},\; \dot{v}_{y,i},\; \dot{\psi}_e,\; \dot{\omega}_{z,e}\bigr) \in \mathbb{R}^{124}
  \label{eq:pinn:rhs-full}
\end{equation}

Key components:
\begin{itemize}
  \item \textbf{Kinematics:} $\dot{x}_i = v_{x,i}$, $\dot{\psi}_e = \omega_{z,e}$
  \item \textbf{Bending moments:} $M_j = EI(\psi_{j+1} - \psi_j)/\Delta\ell$
  \item \textbf{RFT friction:} $\mathbf{F}_i^{\text{fric}} = -c_t\mathbf{v}_{\text{tan},i} - c_n\mathbf{v}_{\text{norm},i}$ with $\varepsilon$-regularized tangents
  \item \textbf{Newton's 2nd law:} $\dot{v}_{x,i} = (F_{x,i}^{\text{fric}} + F_{x,i}^{\text{elastic}})/m_i$
\end{itemize}

\subsubsection{Physics-Regularized Loss}
\label{sec:pinn:reg-loss}

For the regularized surrogate:
\begin{equation}
  \mathcal{L}^{(n)}(\theta) = \mathcal{L}_{\mathrm{data}}(\theta) + \lambda\,\sigma_{\mathrm{reg}}(n)\,\mathcal{L}_{\mathrm{reg}}(\theta)
  \label{eq:pinn:reg-total-loss}
\end{equation}
with piecewise-linear warmup (20\% data-only).

\subsubsection{DD-PINN Loss and Training}
\label{sec:pinn:ddpinn-training}

The DD-PINN training loss at epoch $n$:
\begin{equation}
  \boxed{%
    \mathcal{L}^{(n)}(\theta) = w_{\mathrm{data}}^{(n)}\,\mathcal{L}_{\mathrm{data}}(\theta)
    + w_{\mathrm{phys}}^{(n)}\,\lambda\,\sigma(n)\,\mathcal{L}_{\mathrm{phys}}(\theta)
  }
  \label{eq:pinn:ddpinn-total}
\end{equation}
where $w_{\mathrm{data}}^{(n)}, w_{\mathrm{phys}}^{(n)}$ are ReLoBRaLo \citep{Bischof2022} adaptive weights, $\lambda$ is fixed physics strength, and $\sigma(n)$ is a curriculum schedule (15\% warmup, 30\% ramp, full physics at 45\%).

\subsection{Method of Lines and Explicit Integration}
\label{app:physics:mol}

The PDE-to-ODE reduction follows two stages:

\paragraph{Stage 1 --- Spatial discretization (method of lines).}
All spatial derivatives $\partial/\partial s$ are replaced by sparse matrix--vector products:
\begin{itemize}
  \item Force gradient: $\partial \mathbf{F}/\partial s \to \mathbf{D}^T \mathbf{F}_e$ (bidiagonal, $O(N)$)
  \item Strain from geometry: $\boldsymbol{\varepsilon}_e = \mathbf{Q}_e^T (\mathbf{x}_{e+1} - \mathbf{x}_e)/\ell_e^{\text{rest}}$
  \item Constitutive law: $\mathbf{F}_e = \mathbf{S}(\boldsymbol{\varepsilon}_e - \boldsymbol{\varepsilon}_e^{\text{rest}})$ (pointwise)
\end{itemize}
After replacement, the PDE becomes the ODE $\dot{\mathbf{s}} = \mathbf{g}(\mathbf{s}, \mathbf{a}, t)$ with $\mathbf{g}$ a closed-form function.

\paragraph{Stage 2 --- Explicit time integration.}
An explicit integrator evaluates $\mathbf{g}$ at the current state and writes the result directly:
$\mathbf{s}^{n+1} = \Phi(\mathbf{s}^n, \Delta t)$.
The unknown appears only on the left-hand side --- no equation to solve.

\paragraph{CFL stability constraint.}
$\Delta t < C\sqrt{\rho A \Delta s^4 / (EI)}$, giving $\Delta t_{\max} \approx 1.7 \times 10^{-3}$\,s.
PyElastica uses $\Delta t = 10^{-3}$\,s (safely below), forcing $n_{\text{sub}} = 500$.

\subsection{DisMech Formulation}
\label{app:physics:dismech}

DisMech uses Discrete Elastic Rods (DER) with implicit backward Euler integration.

\paragraph{Backward Euler formulation.}
\begin{subequations}
\label{eq:dismech-backward-euler}
\begin{align}
  \mathbf{v}_{n+1} &= \mathbf{v}_n + \Delta t\,\mathbf{M}^{-1}\mathbf{F}(\mathbf{q}_{n+1},\, \mathbf{v}_{n+1}), \\
  \mathbf{q}_{n+1} &= \mathbf{q}_n + \Delta t\,\mathbf{v}_{n+1},
\end{align}
\end{subequations}

\paragraph{Nonlinear residual and Newton solver.}
\begin{equation}
  \mathbf{g}(\mathbf{q}_{n+1})
    \coloneqq \mathbf{M}\bigl(\mathbf{q}_{n+1} - 2\mathbf{q}_n + \mathbf{q}_{n-1}\bigr)
    - \Delta t^2\,\mathbf{F}(\mathbf{q}_{n+1})
    = \mathbf{0}
  \label{eq:dismech-residual}
\end{equation}
solved iteratively: $\mathbf{q}^{(k+1)} = \mathbf{q}^{(k)} - \mathbf{J}^{-1}\mathbf{g}(\mathbf{q}^{(k)})$ with convergence criteria $\|\mathbf{g}\| < 10^{-4}$ or $\|\delta\mathbf{q}\| < 10^{-2}$, max 25 iterations.

\paragraph{Discrete curvature.}
\begin{equation}
  \boldsymbol{\kappa}_i
    = \frac{2\,\mathbf{e}_{i-1} \times \mathbf{e}_i}
           {|\mathbf{e}_{i-1}|\,|\mathbf{e}_i| + \mathbf{e}_{i-1} \cdot \mathbf{e}_i}
  \label{eq:dismech-curvature-binormal}
\end{equation}
